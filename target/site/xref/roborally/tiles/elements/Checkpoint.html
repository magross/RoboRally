28          if (parameters.length() == 1 && Character.isDigit(parameters.charAt(0))) {
29              number = Integer.parseInt(parameters) - 1;
30              addEffect(Timing.ROBOT_MOVE, new ArchiveEffect(1000, getTile()));
31              addEffect(Timing.ROBOT_MOVE, new ProgressEffect(1000, getTile(), number));
32              addEffect(Timing.END_OF_TURN, new RepairEffect(1000, getTile()));

35              return false;
