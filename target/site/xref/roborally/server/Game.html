115     public Player[] getActivePlayers() {
116         Player[] result;
117         synchronized (players) {
118             int count = players.size();
119             for (Player player : players.elements()) {
120                 if (player.getGameState() != Player.GameState.ACTIVE) {
121                     count--;
122                 }
123             }
124             result = new Player[count];
125             count = 0;
126             for (Player player : players.elements()) {
127                 if (player.getGameState() == Player.GameState.ACTIVE) {
128                     result[count++] = player;
129                 }
130             }
131         }
132         return result;
133     }

160     public boolean join(Player player) {
161         synchronized (players) {
162             if (scenario.getMaximumNumberOfPlayers() > players.size() && !running) {
163                 players.add(player);
164                 player.setGame(this);
165                 player.setState(PlayerState.PLAYING);
166                 thread.manager.broadcast(PLAYER_JOINED, player.getName());
167                 return true;
168             } else {
169                 return false;
170             }
171         }
172     }

174     public void leave(Player player) {
175         synchronized (players) {
176             if (players.contains(player)) {
177                 players.remove(player);
178                 player.setState(PlayerState.REGISTERED);
179                 player.setGame(null);
180                 thread.manager.broadcast(PLAYER_LEFT, player.getName());
181             }
182             if (players.size() == 0) {
183                 server.stopGame(name);
184                 thread.stop();
185             }
186         }
187     }

295         private void respawnDestroyedRobots() {
296             for (Player player : getActivePlayers()) {
297                 Robot robot = player.getRobot();
298                 if (robot.isDestroyed()) {
299                     if (player.getLives() > 0) {
300                         Tile spawnPoint = requestAndWaitForChoice(player, Type.SPAWN_TILE, robot.getArchive().getPossibleSpawnPoints());
301                         robot.setTile(spawnPoint);
302                         spawnPoint.setRobot(robot);
303                         Direction direction = requestAndWaitForChoice(player, Type.SPAWN_DIRECTION, spawnPoint.getPossibleSpawnDirections());
304                         robot.setDirection(direction);
305                         robot.setCurrentHealth(rules.getHealthOnRespawn(robot));
306                         robot.setDestroyed(false);
307                         player.setLives(player.getLives() - 1);
308                     }
309                 }
310             }

348             for (Player player : getActivePlayers()) {
349                 Robot robot = player.getRobot();
350                 if (!robot.isPoweredDown()) {
351                     List<ProgramCard> cards = manager.getChoice(player, Type.PROGRAMMING).getChosenObjects();
352                     int indexX = 0;

378         private List<Player> executeProgramCards(Phase phase) {
379             LinkedList<ProgramCard> queue = new LinkedList<ProgramCard>();
380             LinkedList<Player> order = new LinkedList<Player>();
381             for (Player player : getActivePlayers()) {
382                 Robot robot = player.getRobot();
383                 if (!robot.isDestroyed() && !robot.isPoweredDown()) {
384                     queue.add(robot.getRegisters()[phase.ordinal()].getCard());
385                 }
386             }
387             Collections.sort(queue);
388             for (ProgramCard card : queue) {
389                 Robot robot = card.getRobot();
390                 order.add(robot.getPlayer());
391                 manager.broadcast(EXECUTING_PROGRAM_CARD, card.toMessage());
392                 robot.executeProgramCard(card.getCommand());
393             }
394             return order;
395         }

397         private List<String> processEffects(Timing timing) {
398             List<String> messages = new LinkedList<String>();
399             PriorityQueue<Effect> effects = new PriorityQueue<Effect>();
400             for (Tile tile : grid.getTiles()) {
401                 effects.addAll(tile.getEffects(timing));
402             }
403             while (!effects.isEmpty()) {
404                 Effect effect = effects.poll();
405                 List<Effect> currentEffects = new LinkedList<Effect>();
406                 currentEffects.add(effect);
407                 while (!effects.isEmpty() && effects.peek().getPriority() == effect.getPriority()) {
408                     Effect nextEffect = effects.poll();
409                     boolean add = true;
410                     for (int i = currentEffects.size() - 1; i >= 0; i--) {
411                         if (nextEffect.isConflictingWith(currentEffects.get(i))) {
412                             currentEffects.remove(i);
413                             add = false;
414                         }
415                     }
416                     if (add) {
417                         currentEffects.add(nextEffect);
418                     }
419                 }
420                 for (Effect e : currentEffects) {
421                     e.prepare();
422                 }
423                 for (Effect e : currentEffects) {
424                     e.activate();
425                     messages.addAll(e.getMessages());
426                 }
427             }
428             return messages;
429         }

431         @Override
432         public void run() {
433             Scenario course = scenario.clone();
434             grid = scenario.getTileGrid();
435             rules = new Ruleset();
436 
437             createDeck();
438             players.shuffle();
439 
440             Player[] p = getPlayers();
441             for (int i = 0; i < p.length; i++) {
442                 p[i].setPosition(i);
443             }
444 
445             int index = 0;
446             for (Player player : getPlayers()) {
447                 Robot robot = new Robot(player);
448                 Tile tile = grid.getStartingPoint(index);
449                 index++;
450                 player.setGameState(Player.GameState.ACTIVE);
451                 player.setLives(rules.getNumberOfLives());
452                 player.setRobot(robot);
453                 robot.setArchive(tile);
454                 robot.setMaximumHealth(rules.getMaximumHealth(robot));
455                 robot.setCurrentHealth(rules.getStartingHealth(robot));
456                 robot.setDestroyed(false);
457                 robot.setDirection(null);
458                 robot.setPoweredDown(false);
459                 robot.setTile(tile);
460                 tile.setRobot(robot);
461             }
462 
463             manager.broadcast(GAME_STARTED, name);
464             manager.broadcast(GAME_PLAYERS, players);
465             manager.broadcast(GAME_STATUS, course.getDetailedMessage());
466 
467             // Let the players determine starting positions for their robots
468             chooseStartingDirections();
469 
470             while (getActivePlayers().length > 0) {
471                 turn++;
472                 manager.broadcast(NEW_TURN, turn);
473 
474                 // Deal with destroyed robots
475                 respawnDestroyedRobots();
476 
477                 // Collect all free program cards and return them to the discard stack
478                 List<ProgramCard> programCards = new LinkedList<ProgramCard>();
479                 for (Player player : getPlayers()) {
480                     Robot robot = player.getRobot();
481                     programCards.addAll(robot.extractProgramCards());
482                 }
483                 deck.discardCards(programCards);
484 
485                 // Ask powered down robots whether they want to stay that way
486                 askForContinuedPowerDowns();
487 
488                 // Deal with power down announcements
489                 processPowerDownAnnouncements();
490 
491                 // Deal program cards
492                 dealProgramCards();
493 
494                 // Wait for the programming to be finished
495                 manager.waitForAllOpenChoices(rules.getTimeout(Type.PROGRAMMING));
496                 preparePrograms();
497                 manager.clearChoices();
498 
499                 // Ask for power down announcements
500                 askForPowerDownAnnouncements();
501 
502                 // Execute the programs
503                 manager.broadcast(EXECUTING_PROGRAMS);
504                 for (Phase phase : Phase.values()) {
505                     manager.broadcast(NEW_PHASE, phase.ordinal() + 1);
506 
507                     // Update the tile state
508                     updateTilePhase(phase);
509 
510                     // Get the commands for the current phase and order them by priority
511                     List<Player> fireOrder = executeProgramCards(phase);
512 
513                     List<String> messages = new LinkedList<String>();
514 
515                     // Move factory elements
516                     messages.addAll(processEffects(Timing.BOARD_ELEMENT_MOVE));
517 
518                     // Fire Robot Lasers
519                     for (Player player : fireOrder) {
520                         Robot robot = player.getRobot();
521                         robot.fireWeapons();
522                     }
523 
524                     // Laser fire
525                     messages.addAll(processEffects(Timing.LASER_FIRE));
526 
527                     // Touch checkpoints
528                     messages.addAll(processEffects(Timing.TOUCH_CHECKPOINT));
529 
530                     for (Player player : getActivePlayers()) {
531                         if (player.getRobot().isDestroyed() && player.getLives() == 0 && player.getGameState() == Player.GameState.ACTIVE) {
532                             player.setGameState(Player.GameState.DESTROYED);
533                             player.setFinalTurn(turn);
534                             destroyedPlayers.add(0, player);
535                             manager.broadcast(PLAYER_DESTROYED, player.getName());
536                         }
537                     }
538 
539                     for (Player player : getActivePlayers()) {
540                         if (player.getRobot().getProgress() == grid.getNumberOfCheckpoints() && player.getGameState() == Player.GameState.ACTIVE) {
541                             player.setGameState(Player.GameState.COMPLETED);
542                             player.setFinalTurn(turn);
543                             arrivedPlayers.add(player);
544                             manager.broadcast(PLAYER_ARRIVED, player.getName());
545                         }
546                     }
547 
548                     // Inform players about important effects
549                     for (String message : messages) {
550                         manager.broadcast(EFFECT_OCCURRED, message);
551                         //System.out.println("Game: " + message);
552                     }
553 
554                     // Inform players about the current robot status
555                     for (Player player : getActivePlayers()) {
556                         Robot robot = player.getRobot();
557                         manager.broadcast(ROBOT_STATUS, robot.toMessage());
558                     }
559                 }
560 
561                 // Process end of turn effects and inform players if neccessary
562                 List<String> messages = processEffects(Timing.END_OF_TURN);
563                 for (String message : messages) {
564                     manager.broadcast(EFFECT_OCCURRED, message);
565                 }
566                 for (Player player : getActivePlayers()) {
567                     Robot robot = player.getRobot();
568                     manager.broadcast(ROBOT_STATUS, robot.toMessage());
569                 }
570             }
571 
572             List<Player> ranking = new LinkedList<Player>();
573             ranking.addAll(arrivedPlayers);
574             Collections.sort(destroyedPlayers, new Comparator<Player>() {
575 
576                 public int compare(Player o1, Player o2) {
577                     int result = o2.getRobot().getProgress() - o1.getRobot().getProgress();
578                     return result;
579                 }
580             });
581             ranking.addAll(destroyedPlayers);
582             manager.broadcast(GAME_OVER, ranking);
583 
584             DateFormat dateFormatter = DateFormat.getDateInstance();
585             DateFormat timeFormatter = DateFormat.getTimeInstance();
586             String date = dateFormatter.format(Calendar.getInstance().getTime());
587             String time = timeFormatter.format(Calendar.getInstance().getTime()).replaceAll(":", ".");
588             BufferedWriter writer = null;
589             try {
590                 File replayFolder = new File("replays");
591                 if (!replayFolder.exists()) {
592                     replayFolder.mkdir();
593                 }
594                 writer = new BufferedWriter(new FileWriter(String.format("replays%1$s%2$s (%3$s %4$s)", File.separator, name, date, time)));
595                 for (String line : manager.getLog()) {
596                     writer.write(line);
597                     writer.newLine();
598                 }
599             } catch (IOException ex) {
600                 Logger.getLogger(Game.class.getName()).log(Level.SEVERE, null, ex);
601             } finally {
602                 if (writer != null) {
603                     try {
604                         writer.flush();
605                         writer.close();
606                     } catch (IOException ex) {
607                         Logger.getLogger(Game.class.getName()).log(Level.SEVERE, null, ex);
608                     }
609                 }
610             }
611 
612             //System.out.println("Players leaving");
613             for (Player player : getPlayers()) {
614                 leave(player);
615             }
616 
617         }
