7   import java.util.EnumMap;
8   import java.util.LinkedList;
9   import java.util.List;
10  import roborally.Timing;

16  public class EffectManager {
17  
18      private EnumMap<Timing, List<Effect>> effects;
19  
20      public EffectManager() {
21          effects = new EnumMap<Timing, List<Effect>>(Timing.class);
22          for (Timing timing : Timing.values()) {
23              effects.put(timing, new LinkedList<Effect>());
24          }
25      }
26  
27      public void add(Timing timing, Effect effect) {
28          int index = 0;
29          while (index < effects.get(timing).size() && effect.getPriority() < effects.get(timing).get(index).getPriority()) {
30              index++;
31          }
32          effects.get(timing).add(index, effect);
33      }
34  
35      public List<Effect> get(Timing timing) {
36          return effects.get(timing);
37      }
